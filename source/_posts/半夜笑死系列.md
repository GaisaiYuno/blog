int notation_frequency(int t)
{
        switch (t)
        {
        /*中音部分:1,2,3,4,5,6,7*/
        case 1:return 523;
        case 2:return 587;
        case 3:return 659;
        case 4:return 698;
        case 5:return 784;

        case 12:return 1568;

        default:return 0;

}
